Файл: api/payment.js code JavaScript const crypto = require('crypto'); export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Method Not Allowed' }); } try { const { amount, clientName, clientEmail, clientPhone } = req.body; // Ці ключі ти маєш додати в налаштуваннях Vercel (Settings -> Environment Variables) const merchantAccount = process.env.WFP_MERCHANT_ACCOUNT; const merchantSecretKey = process.env.WFP_MERCHANT_SECRET; // Зміни на домен свого сайту на Vercel (без слеша вкінці) const merchantDomainName = 'https://fut.vercel.app'; const returnUrl = `${merchantDomainName}/thank-you.html`; if (!merchantAccount || !merchantSecretKey) { throw new Error("WayForPay keys are missing in Vercel Environment Variables"); } const orderReference = `ORDER_${Date.now()}`; const orderDate = Math.round(Date.now() / 1000); const currency = 'UAH'; const productName =['Персональна діагностика від Ілони Солдатюк']; // Обов'язково масив const productCount = [1]; // Обов'язково масив const productPrice = [amount]; // Обов'язково масив // Формуємо рядок для підпису відповідно до документації WayForPay const signString =[ merchantAccount, merchantDomainName, orderReference, orderDate, amount, currency, productName[0], productCount[0], productPrice[0] ].join(';'); // Генеруємо HMAC_MD5 підпис const signature = crypto.createHmac('md5', merchantSecretKey).update(signString).digest('hex'); // Повертаємо всі необхідні поля на фронтенд, щоб він створив HTML-форму res.status(200).json({ merchantAccount, merchantDomainName, orderReference, orderDate, amount, currency, productName, productCount, productPrice, clientFirstName: clientName, clientEmail: clientEmail, clientPhone: clientPhone, returnUrl, merchantSignature: signature }); } catch (error) { console.error("Payment API Error:", error); res.status(500).json({ error: "Internal Server Error" }); } } 3.
